593  std::string get_name(bool positional = false, //<[input] Show the positional name
594  bool all_options = false //<[input] Show every option
595  ) const {
596 
597  if(all_options) {
598 
599  std::vector<std::string> name_list;
600 
602  if((positional && pname_.length()) || (snames_.empty() && lnames_.empty()))
603  name_list.push_back(pname_);
604  if((get_items_expected() == 0) && (!fnames_.empty())) {
605  for(const std::string &sname : snames_) {
606  name_list.push_back("-" + sname);
607  if(check_fname(sname)) {
608  name_list.back() += "{" + get_flag_value(sname, "") + "}";
609  }
610  }
611 
612  for(const std::string &lname : lnames_) {
613  name_list.push_back("--" + lname);
614  if(check_fname(lname)) {
615  name_list.back() += "{" + get_flag_value(lname, "") + "}";
616  }
617  }
618  } else {
619  for(const std::string &sname : snames_)
620  name_list.push_back("-" + sname);
621 
622  for(const std::string &lname : lnames_)
623  name_list.push_back("--" + lname);
624  }
625 
626  return detail::join(name_list);
627 
628  } else {
629 
630  // This returns the positional name no matter what
631  if(positional)
632  return pname_;
633 
634  // Prefer long name
635  else if(!lnames_.empty())
636  return std::string("--") + lnames_[0];
637 
638  // Or short name if no long name
639  else if(!snames_.empty())
640  return std::string("-") + snames_[0];
641 
642  // If positional is the only name, it's okay to use that
643  else
644  return pname_;
645  }
646  }

653  void run_callback() {
654 
655  callback_run_ = true;
656 
657  // Run the validators (can change the string)
658  if(!validators_.empty()) {
659  for(std::string &result : results_)
660  for(const std::function<std::string(std::string &)> &vali : validators_) {
661  std::string err_msg;
662 
663  try {
664  err_msg = vali(result);
665  } catch(const ValidationError &err) {
666  throw ValidationError(err.what(), get_name());
667  }
668 
669  if(!err_msg.empty())
670  throw ValidationError(get_name(), err_msg);
671  }
672  }
673  if(!(callback_)) {
674  return;
675  }
676  bool local_result;
677 
678  // Num items expected or length of vector, always at least 1
679  // Only valid for a trimming policy
680  int trim_size =
681  std::min<int>(std::max<int>(std::abs(get_items_expected()), 1), static_cast<int>(results_.size()));
682 
683  // Operation depends on the policy setting
684  if(multi_option_policy_ == MultiOptionPolicy::TakeLast) {
685  // Allow multi-option sizes (including 0)
686  results_t partial_result{results_.end() - trim_size, results_.end()};
687  local_result = !callback_(partial_result);
688 
689  } else if(multi_option_policy_ == MultiOptionPolicy::TakeFirst) {
690  results_t partial_result{results_.begin(), results_.begin() + trim_size};
691  local_result = !callback_(partial_result);
692 
693  } else if(multi_option_policy_ == MultiOptionPolicy::Join) {
694  results_t partial_result = {detail::join(results_, "\n")};
695  local_result = !callback_(partial_result);
696 
697  } else {
698  // Exact number required
699  if(get_items_expected() > 0) {
700  if(results_.size() != static_cast<size_t>(get_items_expected()))
701  throw ArgumentMismatch(get_name(), get_items_expected(), results_.size());
702  // Variable length list
703  } else if(get_items_expected() < 0) {
704  // Require that this be a multiple of expected size and at least as many as expected
705  if(results_.size() < static_cast<size_t>(-get_items_expected()) ||
706  results_.size() % static_cast<size_t>(std::abs(get_type_size())) != 0)
707  throw ArgumentMismatch(get_name(), get_items_expected(), results_.size());
708  }
709  local_result = !callback_(results_);
710  }
711 
712  if(local_result)
713  throw ConversionError(get_name(), results_);
714  }

738  bool check_name(std::string name) const {
739 
740  if(name.length() > 2 && name[0] == '-' && name[1] == '-')
741  return check_lname(name.substr(2));
742  else if(name.length() > 1 && name.front() == '-')
743  return check_sname(name.substr(1));
744  else {
745  std::string local_pname = pname_;
746  if(ignore_underscore_) {
747  local_pname = detail::remove_underscore(local_pname);
748  name = detail::remove_underscore(name);
749  }
750  if(ignore_case_) {
751  local_pname = detail::to_lower(local_pname);
752  name = detail::to_lower(name);
753  }
754  return name == local_pname;
755  }

774  std::string get_flag_value(std::string name, std::string input_value) const {
775  static const std::string trueString{"true"};
776  static const std::string falseString{"false"};
777  static const std::string emptyString{"{}"};
778  // check for disable flag override_
779  if(disable_flag_override_) {
780  if(!((input_value.empty()) || (input_value == emptyString))) {
781  auto default_ind = detail::find_member(name, fnames_, ignore_case_, ignore_underscore_);
782  if(default_ind >= 0) {
783  if(default_flag_values_[default_ind].second != input_value) {
784  throw(ArgumentMismatch::FlagOverride(name));
785  }
786  } else {
787  if(input_value != trueString) {
788  throw(ArgumentMismatch::FlagOverride(name));
789  }
790  }
791  }
792  }
793  auto ind = detail::find_member(name, fnames_, ignore_case_, ignore_underscore_);
794  if((input_value.empty()) || (input_value == emptyString)) {
795  return (ind < 0) ? trueString : default_flag_values_[ind].second;
796  }
797  if(ind < 0) {
798  return input_value;
799  }
800  if(default_flag_values_[ind].second == falseString) {
801  try {
802  auto val = detail::to_flag_value(input_value);
803  return (val == 1) ? falseString : (val == (-1) ? trueString : std::to_string(-val));
804  } catch(const std::invalid_argument &) {
805  return input_value;
806  }
807  } else {
808  return input_value;
809  }
810  }

846  template <typename T,
847  enable_if_t<!is_vector<T>::value && !std::is_const<T>::value, detail::enabler> = detail::dummy>
848  void results(T &output) const {
849  bool retval;
850  if(results_.empty()) {
851  retval = detail::lexical_cast(defaultval_, output);
852  } else if(results_.size() == 1) {
853  retval = detail::lexical_cast(results_[0], output);
854  } else {
855  switch(multi_option_policy_) {
856  case MultiOptionPolicy::TakeFirst:
857  retval = detail::lexical_cast(results_.front(), output);
858  break;
859  case MultiOptionPolicy::TakeLast:
860  default:
861  retval = detail::lexical_cast(results_.back(), output);
862  break;
863  case MultiOptionPolicy::Throw:
864  throw ConversionError(get_name(), results_);
865  case MultiOptionPolicy::Join:
866  retval = detail::lexical_cast(detail::join(results_), output);
867  break;
868  }
869  }
870  if(!retval) {
871  throw ConversionError(get_name(), results_);
872  }
873  }
875  template <typename T> void results(std::vector<T> &output) const {

881  retval &= detail::lexical_cast(elem, output.back());
882  }
883 
884  if(!retval) {
885  throw ConversionError(get_name(), results_);
886  }
887  }
