20 inline std::string ini_join(std::vector<std::string> args) {
21  std::ostringstream s;
22  size_t start = 0;
23  for(const auto &arg : args) {
24  if(start++ > 0)
25  s << " ";
26 
27  auto it = std::find_if(arg.begin(), arg.end(), [](char ch) { return std::isspace<char>(ch, std::locale()); });
28  if(it == arg.end())
29  s << arg;
30  else if(arg.find_first_of('\"') == std::string::npos)
31  s << '\"' << arg << '\"';
32  else
33  s << '\'' << arg << '\'';
34  }
35 
36  return s.str();
37 }

98  std::vector<ConfigItem> from_config(std::istream &input) const override {
99  std::string line;
100  std::string section = "default";
101 
102  std::vector<ConfigItem> output;
103 
104  while(getline(input, line)) {
105  std::vector<std::string> items_buffer;
106 
107  detail::trim(line);
108  size_t len = line.length();
109  if(len > 1 && line[0] == '[' && line[len - 1] == ']') {
110  section = line.substr(1, len - 2);
111  } else if(len > 0 && line[0] != ';') {
112  output.emplace_back();
113  ConfigItem &out = output.back();
114 
115  // Find = in string, split and recombine
116  auto pos = line.find('=');
117  if(pos != std::string::npos) {
118  out.name = detail::trim_copy(line.substr(0, pos));
119  std::string item = detail::trim_copy(line.substr(pos + 1));
120  items_buffer = detail::split_up(item);
121  } else {
122  out.name = detail::trim_copy(line);
123  items_buffer = {"ON"};
124  }
125 
126  if(detail::to_lower(section) != "default") {
127  out.parents = {section};
128  }
129 
130  if(out.name.find('.') != std::string::npos) {
131  std::vector<std::string> plist = detail::split(out.name, '.');
132  out.name = plist.back();
133  plist.pop_back();
134  out.parents.insert(out.parents.end(), plist.begin(), plist.end());
135  }
136 
137  out.inputs.insert(std::end(out.inputs), std::begin(items_buffer), std::end(items_buffer));
138  }
139  }
140  return output;
141  }
