578  std::string get_name(bool positional = false, //<[input] Show the positional name
579  bool all_options = false //<[input] Show every option
580  ) const {
581 
582  if(all_options) {
583 
584  std::vector<std::string> name_list;
585 
587  if((positional && pname_.length()) || (snames_.empty() && lnames_.empty()))
588  name_list.push_back(pname_);
589  if((get_items_expected() == 0) && (!fnames_.empty())) {
590  for(const std::string &sname : snames_) {
591  name_list.push_back("-" + sname);
592  if(check_fname(sname)) {
593  name_list.back() += "{" + get_flag_value(sname, "") + "}";
594  }
595  }
596 
597  for(const std::string &lname : lnames_) {
598  name_list.push_back("--" + lname);
599  if(check_fname(lname)) {
600  name_list.back() += "{" + get_flag_value(lname, "") + "}";
601  }
602  }
603  } else {
604  for(const std::string &sname : snames_)
605  name_list.push_back("-" + sname);
606 
607  for(const std::string &lname : lnames_)
608  name_list.push_back("--" + lname);
609  }
610 
611  return detail::join(name_list);
612 
613  } else {
614 
615  // This returns the positional name no matter what
616  if(positional)
617  return pname_;
618 
619  // Prefer long name
620  else if(!lnames_.empty())
621  return std::string("--") + lnames_[0];
622 
623  // Or short name if no long name
624  else if(!snames_.empty())
625  return std::string("-") + snames_[0];
626 
627  // If positional is the only name, it's okay to use that
628  else
629  return pname_;
630  }
631  }

638  void run_callback() {
639 
640  callback_run_ = true;
641 
642  // Run the validators (can change the string)
643  if(!validators_.empty()) {
644  for(std::string &result : results_)
645  for(const std::function<std::string(std::string &)> &vali : validators_) {
646  std::string err_msg;
647 
648  try {
649  err_msg = vali(result);
650  } catch(const ValidationError &err) {
651  throw ValidationError(err.what(), get_name());
652  }
653 
654  if(!err_msg.empty())
655  throw ValidationError(get_name(), err_msg);
656  }
657  }
658  if(!(callback_)) {
659  return;
660  }
661  bool local_result;
662 
663  // Num items expected or length of vector, always at least 1
664  // Only valid for a trimming policy
665  int trim_size =
666  std::min<int>(std::max<int>(std::abs(get_items_expected()), 1), static_cast<int>(results_.size()));
667 
668  // Operation depends on the policy setting
669  if(multi_option_policy_ == MultiOptionPolicy::TakeLast) {
670  // Allow multi-option sizes (including 0)
671  results_t partial_result{results_.end() - trim_size, results_.end()};
672  local_result = !callback_(partial_result);
673 
674  } else if(multi_option_policy_ == MultiOptionPolicy::TakeFirst) {
675  results_t partial_result{results_.begin(), results_.begin() + trim_size};
676  local_result = !callback_(partial_result);
677 
678  } else if(multi_option_policy_ == MultiOptionPolicy::Join) {
679  results_t partial_result = {detail::join(results_, "\n")};
680  local_result = !callback_(partial_result);
681 
682  } else {
683  // Exact number required
684  if(get_items_expected() > 0) {
685  if(results_.size() != static_cast<size_t>(get_items_expected()))
686  throw ArgumentMismatch(get_name(), get_items_expected(), results_.size());
687  // Variable length list
688  } else if(get_items_expected() < 0) {
689  // Require that this be a multiple of expected size and at least as many as expected
690  if(results_.size() < static_cast<size_t>(-get_items_expected()) ||
691  results_.size() % static_cast<size_t>(std::abs(get_type_size())) != 0)
692  throw ArgumentMismatch(get_name(), get_items_expected(), results_.size());
693  }
694  local_result = !callback_(results_);
695  }
696 
697  if(local_result)
698  throw ConversionError(get_name(), results_);
699  }

756  std::string get_flag_value(std::string name, std::string input_value) const {
757  static const std::string trueString{"true"};
758  static const std::string falseString{"false"};
759  static const std::string emptyString{"{}"};
760  // check for disable flag override_
761  if(disable_flag_override_) {
762  if(!((input_value.empty()) || (input_value == emptyString))) {
763  auto default_ind = detail::find_member(name, fnames_, ignore_case_, ignore_underscore_);
764  if(default_ind >= 0) {
765  if(default_flag_values_[default_ind].second != input_value) {
766  throw(ArgumentMismatch::FlagOverride(name));
767  }
768  } else {
769  if(input_value != trueString) {
770  throw(ArgumentMismatch::FlagOverride(name));
771  }
772  }
773  }
774  }
775  auto ind = detail::find_member(name, fnames_, ignore_case_, ignore_underscore_);
776  if((input_value.empty()) || (input_value == emptyString)) {
777  return (ind < 0) ? trueString : default_flag_values_[ind].second;
778  }
779  if(ind < 0) {
780  return input_value;
781  }
782  if(default_flag_values_[ind].second == falseString) {
783  try {
784  auto val = detail::to_flag_value(input_value);
785  return (val == 1) ? falseString : (val == (-1) ? trueString : std::to_string(-val));
786  } catch(const std::invalid_argument &) {
787  return input_value;
788  }
789  } else {
790  return input_value;
791  }
792  }

824  template <typename T,
825  enable_if_t<!is_vector<T>::value && !std::is_const<T>::value, detail::enabler> = detail::dummy>
826  void results(T &output) const {
827  bool retval;
828  if(results_.empty()) {
829  retval = detail::lexical_cast(defaultval_, output);
830  } else if(results_.size() == 1) {
831  retval = detail::lexical_cast(results_[0], output);
832  } else {
833  switch(multi_option_policy_) {
834  case MultiOptionPolicy::TakeFirst:
835  retval = detail::lexical_cast(results_.front(), output);
836  break;
837  case MultiOptionPolicy::TakeLast:
838  default:
839  retval = detail::lexical_cast(results_.back(), output);
840  break;
841  case MultiOptionPolicy::Throw:
842  throw ConversionError(get_name(), results_);
843  case MultiOptionPolicy::Join:
844  retval = detail::lexical_cast(detail::join(results_), output);
845  break;
846  }
847  }
848  if(!retval) {
849  throw ConversionError(get_name(), results_);
850  }
851  }
853  template <typename T> void results(std::vector<T> &output, char delim = '\0') const {
854  output.clear();
855  bool retval = true;
856 
857  for(const auto &elem : results_) {
858  if(delim != '\0') {
859  for(const auto &var : CLI::detail::split(elem, delim)) {
860  if(!var.empty()) {
861  output.emplace_back();
862  retval &= detail::lexical_cast(var, output.back());
863  }
864  }
865  } else {
866  output.emplace_back();
867  retval &= detail::lexical_cast(elem, output.back());
868  }
869  }
870 
871  if(!retval) {
872  throw ConversionError(get_name(), results_);
873  }
874  }
